% TABLE COMMANDS
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

% Insert table from TEX file: caption, label, file
\newcommand{\tabelle}[3]{%
    \begin{table}[H]
    \centering
    \singlespacing
    \input{src/#3}
    \caption{#1}
    \label{tab:#2}
    \end{table}}
    
% Insert table from TEX file in appendix: caption, file
\newcommand{\tabelleAnhang}[2]{%
    \begin{table}[H]
    \centering
    \singlespacing
    \input{src/Tabellen/#2}
    \caption{#1}
    \end{table}}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% FIGURE COMMANDS
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

% Insert graph: caption, label, file
\newcommand{\graph}[3]{%
        \begin{figure}[H]
            \centering
            \input{src/#3}
            \caption{#1}
            \label{graph:#2}
        \end{figure}
}

% Insert screenshot: caption, label, file, title
\newcommand{\screenshot}[4]{%
        \begin{figure}[H]
            \centering
            \textbf{#4}\par\medskip
            \adjustbox{max width=\linewidth}{
                \includegraphics[scale=0.3]{src/#3}
            }
            \caption{#1}
            \label{graph:#2}
        \end{figure}
}

% Insert circuit: caption, label, file
\newcommand{\circuit}[3]{%
        \begin{figure}[H]
            \centering
            \adjustbox{max width=\linewidth}{
                \input{src/#3}
            }
            \caption{#1}
            \label{circuit:#2}
        \end{figure}
}
